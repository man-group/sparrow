            GIT_PROGRESS TRUE

            Catch2
            GIT_SHALLOW TRUE
            GIT_REPOSITORY https://github.com/catchorg/Catch2.git
            GIT_TAG ${CATCH2_VERSION}
            GIT_PROGRESS TRUE
            SYSTEM
            EXCLUDE_FROM_ALL)

        message(STATUS "\tâœ… Fetched Catch2 ${CATCH2_VERSION}")

    if(NOT nanoarrow_FOUND)
        set(NANOARROW_VERSION "0.6.0")

            nanoarrow
            GIT_SHALLOW TRUE
            GIT_REPOSITORY https://github.com/apache/arrow-nanoarrow.git
            GIT_TAG apache-arrow-nanoarrow-${NANOARROW_VERSION}
            GIT_PROGRESS TRUE
            SYSTEM
            EXCLUDE_FROM_ALL)
        FetchContent_MakeAvailable(nanoarrow)
        target_compile_options(nanoarrow PRIVATE -w)
        message(STATUS "\tâœ… Fetched nanoarrow ${NANOARROW_VERSION}")
    endif()
endif()
